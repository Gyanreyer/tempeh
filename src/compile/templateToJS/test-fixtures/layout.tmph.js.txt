import {TmphHTML, html} from "tempeh/render";

export default class Layout {
  /**
   * @param {Object} renderConfig
   * @param {unknown} renderConfig.slot
   *
   * @returns {TmphHTML}
   */
  render({ slot }) {
    return html`<!DOCTYPE html>
      <html lang="en">
        <head>
          <meta charset="UTF-8" />
          <meta name="viewport" content="width=device-width" />
        </head>
        <body>
          ${slot}
        </body>
      </html>`;
  }
};
